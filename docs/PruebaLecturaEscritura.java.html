import fundamentos.*;
public class PruebaLecturaEscritura {

   public static void main (String[] args) {

      Lectura lec = new Lectura ("Prueba de Lectura");

      int i;
      double d;
      String s;

      lec.creaEntrada("Nombre","Pedro Lacasa...................");
      lec.creaEntrada("Edadeeeeeeeeeeeeeeeeeeeeeeeeeeees",37);
      lec.creaEntrada("Capacidad pulmonar (l)",4.3);
      lec.esperaYCierra("Introduce datos y pulsa aceptar");
      s=lec.leeString("Nombre");
      i=lec.leeInt("Edadeeeeeeeeeeeeeeeeeeeeeeeeeeees");
      d=lec.leeDouble("Capacidad pulmonar (l)");

      Escritura esc = new Escritura ("Prueba de Escritura");

      esc.insertaValor("Nombre",s);
      esc.insertaValor("Edadeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeees",i);
      esc.insertaValor("Capacidad pulmonar (l)",d);
      esc.espera();
      System.exit(0);
   }

}
