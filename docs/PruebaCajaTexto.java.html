import fundamentos.*;
public class PruebaCajaTexto {

   public static void main (String[] args) {

      CajaTexto caja = new CajaTexto ("Prueba de Caja de Texto",20,50);

      caja.println("Escribe un numero entero, un real, y varias lineas");
      caja.espera();
      System.out.println("primera linea="+caja.leeString());
      caja.avanzaLinea();
      System.out.println("num entero   ="+caja.leeInt());
      caja.avanzaLinea();
      System.out.println("num real     ="+caja.leeDouble());
      caja.avanzaLinea();
      while (caja.hayMas()) {
	System.out.println("linea ="+caja.leeString());
	caja.avanzaLinea();
	System.out.println("   hay mas="+caja.hayMas());
      }
      caja.borra();
      caja.println("Ahora mete una linea con dos enteros");
      caja.println("y otra con dos reales, y pulsa aceptar");
      caja.esperaYCierra();
      caja.avanzaLinea();
      caja.avanzaLinea();
      System.out.println("Entero 1: "+caja.leeInt(0));
      System.out.println("Entero 2: "+caja.leeInt(1));
      caja.avanzaLinea();
      System.out.println("Real   1: "+caja.leeDouble(0));
      System.out.println("Real   2: "+caja.leeDouble(1));
      
      System.exit(0);
   }

}
