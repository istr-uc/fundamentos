import fundamentos.*;
public class Prueba {

   public static void main (String[] args) {

      CajaTexto caja = new CajaTexto ("Prueba de Caja de Texto",20,50);
      caja.espera();
      System.out.println("num real     ="+caja.leeString());
      caja.avanzaLinea();
      System.out.println("num entero   ="+caja.leeInt());
      caja.avanzaLinea();
      while (caja.hayMas()) {
      	System.out.println("linea ="+caja.leeString());
      	caja.avanzaLinea();
     	System.out.println("   hay mas="+caja.hayMas());
      }
      System.exit(0);
   }

}
