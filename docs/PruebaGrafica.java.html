import fundamentos.*;
public class PruebaGrafica {

   public static void main (String[] args) {

      Grafica g = new Grafica ("Seno y Coseno","x","y");
      double x;

      float r=6.0f;

      // El primer grafico
      g.ponSimbolo(true);
      g.ponColor(Grafica.azul);
      g.ponTitulo("Seno");
      for (int i=0; i<=100; i++) {
	 x = i/10.0;
	 g.inserta(x,Math.sin(x));
      }

      // El segundo grafico
      g.otraGrafica();
      g.ponColor(Grafica.rojo);
      g.ponTitulo("Coseno");
      for (int i=0; i<=100; i++) {
	 x = i/10.0;
	 g.inserta(x,Math.cos(x));
      }
      g.pinta();

      System.exit(0);
   }

}
